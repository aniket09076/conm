#include <stdio.h>

#define SIZE 10

// Function to perform Gauss-Jordan elimination
void gaussJordan(float matrix[SIZE][SIZE], int n) {
    int i, j, k;
    float ratio;

    for (i = 0; i < n; i++) {
        if (matrix[i][i] == 0.0) {
            printf("Mathematical Error: Division by zero detected.\n");
            return;
        }
        for (j = 0; j < n; j++) {
            if (i != j) {
                ratio = matrix[j][i] / matrix[i][i];
                for (k = 0; k <= n; k++) {
                    matrix[j][k] -= ratio * matrix[i][k];
                }
            }
        }
    }

    // Obtain solution
    for (i = 0; i < n; i++) {
        matrix[i][n] = matrix[i][n] / matrix[i][i];
        matrix[i][i] = 1;
    }
}

int main() {
    float matrix[SIZE][SIZE];
    int i, j, n;

    // Input number of variables
    printf("Enter number of variables: ");
    scanf("%d", &n);

    // Input the augmented matrix
    printf("Enter the augmented matrix row-wise:\n");
    for (i = 0; i < n; i++) {
        for (j = 0; j <= n; j++) {
            scanf("%f", &matrix[i][j]);
        }
    }

    // Perform Gauss-Jordan elimination
    gaussJordan(matrix, n);

    // Output the result
    printf("\nThe solution is:\n");
    for (i = 0; i < n; i++) {
        printf("x%d = %.3f\n", i + 1, matrix[i][n]);
    }

    return 0;
}
